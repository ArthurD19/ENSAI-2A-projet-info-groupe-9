@startuml
left to right direction

actor "Utilisateur"      as user
actor "Administrateur"   as admin
actor "Joueur"           as joueur

node "Serveur Poker" as Titre {
   package "Fonctions Utilisateur" {
    usecase "Créer compte / Se connecter" as authentification
    usecase "Consulter son solde"         as consultation_solde
    usecase "Quitter"                     as quitter_application
  }

  package "Fonctions Joueur" {
    usecase "Rejoindre une table"         as rejoindre_table
    usecase "Jouer"                       as actions_poker
    usecase "Quitter la partie"           as quitter_partie
  }

  package "Fonctions Admin" {
    usecase "Superviser parties"          as supervision
    usecase "Bannir joueur"               as bannissement
    usecase "Créer table"                 as creation_table
    usecase "Fermer table"                as fermeture_table
    usecase "Créditer compte"             as creditation
    usecase "Retirer crédit"              as penalites
  }
  
}

user --> authentification
user --> consultation_solde
user --> quitter_application

joueur --> quitter_partie
joueur --> rejoindre_table
joueur --> actions_poker

admin --> supervision
admin --> bannissement
admin --> creation_table
admin --> fermeture_table
admin --> creditation
admin --> penalites


@enduml
